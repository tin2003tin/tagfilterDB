74inline uint32_t MurmurHash::Hash(const char *data, size_t n, uint32_t seed) {
75 const uint32_t m = 0xc6a4a793;
76 const uint32_t r = 24;
77 const char *limit = data + n;
78 uint32_t h = seed ^ (n * m);
79
80 // Process 4 bytes at a time
81 while (data + 4 <= limit) {
82 uint32_t w = Decode32(data);
83 data += 4;
84 h += w;
85 h *= m;
86 h ^= (h >> 16);
87 }
88
89 // Handle remaining bytes
90 switch (limit - data) {
91 case 3:
92 h += static_cast<uint8_t>(data[2]) << 16;
93 FALLTHROUGH_INTENDED;
94 case 2:
95 h += static_cast<uint8_t>(data[1]) << 8;
96 FALLTHROUGH_INTENDED;
97 case 1:
98 h += static_cast<uint8_t>(data[0]);
99 h *= m;
100 h ^= (h >> r);
101 break;
102 }

105}
