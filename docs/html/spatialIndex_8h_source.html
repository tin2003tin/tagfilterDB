375 int SelectBestSubNode(const BND &r_box, Node *p_node) {
376 int bestIndex = -1;
377 AreaType bestIncr = std::numeric_limits<AreaType>::max();
378 AreaType bestArea = std::numeric_limits<AreaType>::max();
379
380 for (int index = 0; index < p_node->m_csize; ++index) {
381 BND t_box = BND::UnionBox(
382 r_box,
383 p_node->m_subNodes[index].m_box); // Compute the union of boxes
384 AreaType area = p_node->m_subNodes[index]
385 .m_box.area(); // Area of the current subnode's box
386 AreaType increase =
387 t_box.area() - area; // Increase in area due to the new box
388
389 // Update best index based on area increase
390 if (increase < bestIncr || (increase == bestIncr && area < bestArea)) {
391 bestIndex = index;
392 bestIncr = increase;
393 bestArea = area;
394 }
395 }
396 return bestIndex;

424 void SplitNode(const SubNode &r_subNode, Node *p_node) {
425 assert(p_node);
426 assert(p_node->m_csize == MaxChildren);
427
428 GroupAssign t_groupAssign; // Holds the assignment of SubNodes to groups
429 SubNode
430 t_overflowBuffer[5]; // Buffer to hold nodes during the split process
431 BND t_boundingBox =
432 p_node->m_subNodes[0].m_box; // Bounding box of the current node
433 AreaType t_overflowBufferArea[MaxChildren + 1]; // Areas of the bounding
434 // boxes in the buffer
435
436 // Copy existing SubNodes and the new SubNode into t_overflowBuffer
437 for (int i_subNode = 0; i_subNode < MaxChildren; i_subNode++) {
438 t_overflowBuffer[i_subNode] = p_node->m_subNodes[i_subNode];
439 t_boundingBox =
440 BND::UnionBox(t_boundingBox, t_overflowBuffer[i_subNode].m_box);
441 t_overflowBufferArea[i_subNode] =
442 t_overflowBuffer[i_subNode].m_box.area();
443 }
444 t_overflowBuffer[MaxChildren] =
445 r_subNode; // Add the new SubNode to the buffer
446 t_boundingBox = BND::UnionBox(
447 t_boundingBox,
448 r_subNode.m_box); // Update the bounding box with the new SubNode
449 t_overflowBufferArea[MaxChildren] =
450 r_subNode.m_box.area(); // Update the area of the new SubNode
451
452 int seed0 = 0, seed1 = 0; // Indices of the seeds for splitting
453 RangeType bestNormalizedSeparation =
454 -std::numeric_limits<RangeType>::infinity(); // Initialize best
455 // separation value
456
457 // Find the best pair of seeds for the split based on bounding box
458 // dimensions
459 for (int dim = 0; dim < Dimensions; ++dim) {
460 int minIndex = 0, maxIndex = 0;
461 RangeType minLower = t_overflowBuffer[0].m_box.min(dim);
462 RangeType maxUpper = t_overflowBuffer[0].m_box.max(dim);
463
464 for (int index = 1; index < MaxChildren + 1; ++index) {
465 // Find the minimum and maximum extents for the current dimension
466 if (t_overflowBuffer[index].m_box.min(dim) < minLower) {
467 minLower = t_overflowBuffer[index].m_box.min(dim);
468 minIndex = index;
469 }
470 if (t_overflowBuffer[index].m_box.max(dim) > maxUpper) {
471 maxUpper = t_overflowBuffer[index].m_box.max(dim);
472 maxIndex = index;
473 }
474 }
475
476 // Calculate normalized separation for the current dimension
477 RangeType denominator = t_boundingBox.max(dim) - t_boundingBox.min(dim);
478 RangeType reciprocal = 1.0 / denominator;
479 RangeType separation = (maxUpper - minLower) * reciprocal;
480
481 // Update best seeds if current separation is better
482 if (separation > bestNormalizedSeparation) {
483 bestNormalizedSeparation = separation;
484 seed0 = minIndex;
485 seed1 = maxIndex;
486 }
487 }
488
489 // Initialize group assignments with the chosen seeds
490 AssignGroup(seed0, 0, t_overflowBuffer[seed0].m_box, t_groupAssign);
491 AssignGroup(seed1, 1, t_overflowBuffer[seed1].m_box, t_groupAssign);
492
493 bool firstTime = true; // Flag for first iteration
494 int chosen = -1; // Index of the chosen SubNode for assignment
495 int betterGroup = -1; // Index of the better group for assignment
496 int group; // Current group being considered
497 AreaType biggestDiff; // Biggest difference in growth between groups
498
499 // Assign remaining SubNodes to groups based on growth criteria
500 while (
501 (t_groupAssign.m_groups[0].m_count + t_groupAssign.m_groups[1].m_count <
502 t_groupAssign.m_size) &&
503 (t_groupAssign.m_groups[0].m_count <
504 (t_groupAssign.m_size - MinChildren)) &&
505 (t_groupAssign.m_groups[1].m_count <
506 (t_groupAssign.m_size - MinChildren))) {
507
508 bool firstTime = true;
509 for (int index = 0; index < t_groupAssign.m_size; ++index) {
510 if (t_groupAssign.m_groupAssign[index] == -1) {
511 // Compute growth of bounding boxes if the current SubNode is
512 // added to each group
513 BND box0 = BND::UnionBox(t_overflowBuffer[index].m_box,
514 t_groupAssign.m_groups[0].m_box);
515 BND box1 = BND::UnionBox(t_overflowBuffer[index].m_box,
516 t_groupAssign.m_groups[1].m_box);
517 AreaType growth0 =
518 box0.area() - t_groupAssign.m_groups[0].m_box.area();
519 AreaType growth1 =
520 box1.area() - t_groupAssign.m_groups[1].m_box.area();
521 AreaType diff = growth1 - growth0;
522
523 // Determine which group would be better for the current SubNode
524 if (diff >= 0) {
525 group = 0;
526 } else {
527 group = 1;
528 diff = -diff;
529 }
530
531 // Update the best assignment based on the biggest difference
532 if (firstTime || diff > biggestDiff) {
533 firstTime = false;
534 biggestDiff = diff;
535 chosen = index;
536 betterGroup = group;
537 } else if ((diff == biggestDiff) &&
538 (t_groupAssign.m_groups[group].m_count <
539 t_groupAssign.m_groups[betterGroup].m_count)) {
540 chosen = index;
541 betterGroup = group;
542 }
543 }
544 }
545
546 // Ensure we found a valid choice
547 assert(!firstTime);
548 // Assign the chosen SubNode to the better group
549 AssignGroup(chosen, betterGroup, t_overflowBuffer[chosen].m_box,
550 t_groupAssign);
551 }
552
553 // Final adjustments if the total number of nodes is less than expected
554 if ((t_groupAssign.m_groups[0].m_count +
555 t_groupAssign.m_groups[1].m_count) < t_groupAssign.m_size) {
556 // Determine which group to assign remaining nodes to
557 if (t_groupAssign.m_groups[0].m_count >=
558 t_groupAssign.m_size - MinChildren) {
559 group = 1;
560 } else {
561 group = 0;
562 }
563 // Assign remaining nodes to the determined group
564 for (int i = 0; i < t_groupAssign.m_size; i++) {
565 if (t_groupAssign.m_groupAssign[i] == -1) {
566 AssignGroup(i, group, t_overflowBuffer[i].m_box, t_groupAssign);
567 }
568 }
569 }
570
571 // Ensure valid group assignments
572 assert((t_groupAssign.m_groups[0].m_count +
573 t_groupAssign.m_groups[1].m_count) == t_groupAssign.m_size);
574 assert(t_groupAssign.m_groups[0].m_count >= MinChildren);
575 assert(t_groupAssign.m_groups[1].m_count >= MinChildren);
576
577 // Clear the current node and create a new node for one group
578 p_node->m_csize = 0;
579 Node *t_node = new Node(0, p_node->m_height);
580 Node *targetNodes[] = {p_node,
581 t_node}; // Nodes to receive the split SubNodes
582
583 // Ensure the number of nodes in the split is correct
584 assert(t_groupAssign.m_size <= MaxChildren + 1);
585
586 for (int index = 0; index < t_groupAssign.m_size; index++) {
587 int groupAssignValue = t_groupAssign.m_groupAssign[index];
588 assert(groupAssignValue == 0 || groupAssignValue == 1);
589
590 // Add each SubNode to the appropriate group node
591 bool nodeSplited =
592 AddSubNode(t_overflowBuffer[index], targetNodes[groupAssignValue]);
593
594 assert(!nodeSplited);
595 }
596
597 // Ensure the split was performed correctly
598 assert((p_node->m_csize + (t_node)->m_csize) == t_groupAssign.m_size);
599 m_nodeBuffer = &t_node;
600 }

609 void AssignGroup(int a_index, int a_group, BND &r_box,
610 GroupAssign &r_groupAssign) {
611 assert(a_index < r_groupAssign.m_size);
612 assert(a_group < 2);
613 assert(r_groupAssign.m_groupAssign[a_index] == -1);
614
615 // Assign the SubNode to the specified group
616 r_groupAssign.m_groupAssign[a_index] = a_group;
617
618 // Update the bounding box for the group
619 if (r_groupAssign.m_groups[a_group].m_count == 0) {
620 // If this is the first SubNode in the group, initialize the bounding
621 // box
622 r_groupAssign.m_groups[a_group].m_box = r_box;
623 } else {
624 // Otherwise, expand the existing bounding box to include the new
625 // SubNode
626 r_groupAssign.m_groups[a_group].m_box =
627 BND::UnionBox(r_groupAssign.m_groups[a_group].m_box, r_box);
628 }
629
630 // Increment the count of SubNodes in the group
631 r_groupAssign.m_groups[a_group].m_count++;
632 }

660 void RecursivelySearchTag(BND &r_target, Node *p_node,
661 ISIndexCallback<SpatialIndex> *callback) {
662 if (p_node == nullptr) {
663 return;
664 }
665
666 // Search through each SubNode in the current node
667 for (int i = 0; i < p_node->m_csize; i++) {
668 // If the node is a leaf and the target box overlaps with the SubNode's
669 // box
670 if (p_node->isLeaf() &&
671 p_node->m_subNodes[i].m_box.isOverlap(r_target)) {
672 // Process the SubNode using the callback
673 bool conti = callback->process(p_node->m_subNodes[i]);
674 // If the callback returns false, stop further processing
675 if (!conti) {
676 break;
677 }
678 }
679 // Recursively search child nodes
680 RecursivelySearchTag(r_target, p_node->m_subNodes[i].m_child, callback);
681 }
