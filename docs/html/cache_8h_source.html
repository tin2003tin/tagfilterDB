230 BucketValueNode* Insert(Key key, Value value, size_t charge = LRUConfig::DEFAULT_CACHE_CHARGE_PER) {
231 if (charge > m_total_charge) {
232 return nullptr;
233 }
234 assert(charge > 0);
235
236 CacheMux m(&m_mux);
237
238 if (isExpand()) {
239 expand(m_cap * LRUConfig::DEFAULT_CAHCE_EXPAND);
240 }
241
242 std::hash<Key> h;
243 size_t u = h(key);
244 BucketValueNode* newNode = new BucketValueNode(key,value,charge);
245
246 size_t index = u % m_cap;
247 BucketNode* prev = &m_data[index];
248
249 while (prev->m_next != nullptr)
250 {
251 if (((BucketValueNode* )(prev->m_next))->m_key == key) {
252 break;
253 }
254 prev = prev->m_next;
255 }
256 if (prev->m_next == nullptr) {
257 prev->m_next = newNode;
258 m_size++;
259 } else {
260 // The key already existed
261 newNode->m_next = prev->m_next->m_next;
262
263 removeList(prev->m_next);
264 m_total_usage -= ((BucketValueNode* )(prev->m_next))->m_charge;
265 delete prev->m_next;
266
267 prev->m_next = newNode;
268 }
269
270 // Remove in OutDated List if excess a total charge
271 while (m_total_usage + charge > m_total_charge
272 && m_outdated_tail->m_lPrev != m_outdated_head) {
273 bool s = removeNode(((BucketValueNode* )(m_outdated_head->m_lNext))->m_key);
274 assert(s);
275 }
276
277 appendToList(newNode, m_inUsed_tail);
278 m_total_usage += charge;
279
280 assert(m_total_usage > 0);
281 assert(m_total_usage <= m_total_charge);
282 ref(newNode);
283 return newNode;
284 }

493 void expand(size_t newCap) {
494 BucketNode* newLoc = new BucketNode[newCap];
495 std::hash<Key> h;
496 // rehash
497 for (size_t i = 0 ; i < m_cap; i++) {
498 BucketValueNode* curr = (BucketValueNode*) m_data[i].m_next;
499 while (curr != nullptr) {
500 BucketValueNode* next = (BucketValueNode*) curr->m_next;
501 curr->m_next = nullptr;
502 size_t u = h(curr->m_key) % newCap;
503 if (newLoc[u].m_next == nullptr) {
504 newLoc[u].m_next = curr;
505 } else {
506 BucketNode* b = newLoc[u].m_next;
507 while (b->m_next != nullptr)
508 {
509 b= b->m_next;
510 }
511 b->m_next = curr;
512 }
513 curr = next;
514 }
515 }
516 delete[] m_data;
517 m_data = newLoc;
518 m_cap = newCap;
519 }

521 bool removeNode(Key key) {
522 std::hash<Key> h;
523 size_t u = h(key) % m_cap;
524 BucketValueNode* curr = (BucketValueNode*) m_data[u].m_next;
525 BucketNode* prev = &m_data[u];
526 while (curr != nullptr && curr->m_key != key) {
527 prev = curr;
528 curr = (BucketValueNode*) curr->m_next;
529 }
530 if (curr == nullptr) {
531 return false;
532 }
533 m_total_usage -= curr->m_charge;
534 m_size--;
535 prev->m_next = curr->m_next;
536 removeList(curr);
537 delete curr;
538 return true;
539 }
