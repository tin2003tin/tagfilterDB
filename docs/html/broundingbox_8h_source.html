367 RangeType overlap(const BROUNDINGBOX_TYPE &other) {
368 RangeType area = static_cast<RangeType>(1.0);
369 for (std::size_t i_axis = 0; area && i_axis < Dimensions; ++i_axis) {
370 const RangeType t_x1 = m_axis[i_axis].first;
371 const RangeType t_x2 = m_axis[i_axis].second;
372 const RangeType t_y1 = other.m_axis[i_axis].first;
373 const RangeType t_y2 = other.m_axis[i_axis].second;
374
375 if (t_x1 < t_y1) {
376 if (t_y1 < t_x2) {
377 if (t_y2 < t_x2)
378 area *= (t_y2 - t_y1);
379 else
380 area *= (t_x2 - t_y1);
381 continue;
382 }
383 } else if (t_x1 < t_y2) {
384 if (t_x2 < t_y2)
385 area *= (t_x2 - t_x1);
386 else
387 area *= (t_y2 - t_x1);
388 continue;
389 }
390
391 return static_cast<RangeType>(0.0);
392 }
